import pandas as pd
import numpy as np
from sklearn.datasets import load_iris
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import OneHotEncoder, StandardScaler

# Load Iris dataset
iris = load_iris()
X = pd.DataFrame(iris.data, columns=iris.feature_names)

# Simulate missing values (for demonstration)
np.random.seed(0)
X.loc[X.sample(frac=0.1).index, 'sepal length (cm)'] = np.nan

# Simulate a categorical column
X['flower_type'] = np.random.choice(['Type A', 'Type B', 'Type C'], size=len(X))

# 1. Impute Missing Values
imputer = SimpleImputer(strategy='mean')
X_numeric_imputed = imputer.fit_transform(X[iris.feature_names]) # FIXED
X_numeric_imputed_df = pd.DataFrame(X_numeric_imputed, columns=iris.feature_names)
print("After Imputation (Missing Values Handled):")
print(X_numeric_imputed_df.head())

# 2. One-Hot Encode Categorical Column
encoder = OneHotEncoder(handle_unknown='ignore')
X_categorical_encoded = encoder.fit_transform(X[['flower_type']])
encoded_df = pd.DataFrame(
    X_categorical_encoded.toarray(),
    columns=encoder.get_feature_names ()
)

print("\nAfter One-Hot Encoding (Categorical Data Handled):")
print(encoded_df.head())
# 3. Standardize Numeric Features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X_numeric_imputed_df)
X_scaled_df = pd.DataFrame(X_scaled, columns=iris.feature_names) # FIXED
print("\nAfter Standardization (Feature Scaling Applied):")
print(X_scaled_df.head())
# Final Processed Dataset
X_processed = pd.concat([X_scaled_df, encoded_df], axis=1)
print("\nFinal Preprocessed Dataset:")
print(X_processed.head())
print("\nFinal shape:", X_processed.shape)
